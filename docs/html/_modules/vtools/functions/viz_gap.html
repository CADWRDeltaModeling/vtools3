"""Visualization harness for neighbor-based gap filling.

This script/module provides:

1) A *data provider* function interface you can implement to supply
   real examples.
2) Utilities to create synthetic gaps in one or both series.
3) A small experiment runner that applies multiple fill methods.
4) Plotting helpers to visualize imputations versus the true series
   and (optionally) prediction intervals.

Usage (as a module)
-------------------

Implement `provide_example_data()` in your own codebase or monkey-patch it
at runtime. It must return `(target, neighbor, regime)` where `regime` may
be `None`.

Example glue code:

>>> from neighbor_fill_viz import (
...     run_gapfill_experiment, plot_imputations, GapSpec, GapStrategy
... )
>>> # Provide your own function in your environment:
>>> def provide_example_data():
...     '''Return (target, neighbor, regime) with DatetimeIndex and same freq.'''
...     raise NotImplementedError
>>> res = run_gapfill_experiment(
...     data_provider=provide_example_data,
...     gap_spec=GapSpec(n_gaps=4, min_len=6, max_len=24, seed=42,
...                      strategy=GapStrategy.TARGET_ONLY),
...     methods=["ols", "lagged_reg", "state_space"],
...     lags=range(0, 4),
...     resample=None,
... )
>>> fig = plot_imputations(**res, show_pi=True)

You can also run this file as a script to execute a minimal smoke test that
uses a synthetic sine-with-lag dataset if `provide_example_data` is not
found in the import path.
"""

def provide_example_data() -> Tuple[pd.Series, pd.Series | pd.DataFrame, Optional[pd.Series]]:
    """Provide a real-world example.

    You should REPLACE this function in your environment.

    Returns
    -------
    (target, neighbor, regime)
        - `target`: pd.Series with DatetimeIndex, possibly complete (we will gap it).
        - `neighbor`: pd.Series or pd.DataFrame with DatetimeIndex; should be aligned
          or alignable to `target` (same frequency preferred).
        - `regime`: Optional categorical pd.Series aligned to the timestamps (e.g.,
          barrier status). If not applicable, return `None`.

    Notes
    -----
    - Ensure both target and neighbor use the same units and a compatible
      frequency. If not, specify a `resample` frequency in the experiment run.
    - This function is intentionally documented rather than implemented so that
      teams can plug in station-specific datasets.
    """
    raise NotImplementedError(
        "Please implement provide_example_data() in your environment and have it "
        "return (target, neighbor, regime)."
    )

class GapStrategy(Enum):
    """Where to create synthetic gaps."""
    TARGET_ONLY = auto()
    BOTH = auto()            # gap target and neighbor on the same windows
    STAGGERED = auto()       # gap target and neighbor on different windows

@dataclass
class GapSpec:
    n_gaps: int = 5
    min_len: int = 6
    max_len: int = 24
    seed: Optional[int] = 123
    strategy: GapStrategy = GapStrategy.TARGET_ONLY
    ensure_room: int = 2  # min number of intact points between gaps

def _choose_gap_windows(n: int, spec: GapSpec) -> List[Tuple[int, int]]:
    rng = np.random.default_rng(spec.seed)
    windows: List[Tuple[int, int]] = []
    attempts = 0
    while len(windows) < spec.n_gaps and attempts < spec.n_gaps * 100:
        attempts += 1
        length = int(rng.integers(spec.min_len, spec.max_len + 1))
        start = int(rng.integers(0, max(1, n - length)))
        end = start + length
        # enforce spacing and non-overlap
        ok = True
        for (s0, e0) in windows:
            if not (end + spec.ensure_room <= s0 or start >= e0 + spec.ensure_room):
                ok = False
                break
        if ok:
            windows.append((start, min(n, end)))
    windows.sort()
    return windows

def apply_gaps(
    target: pd.Series,
    neighbor: pd.Series | pd.DataFrame,
    spec: GapSpec,
) -> Tuple[pd.Series, pd.Series | pd.DataFrame, Dict[str, List[Tuple[pd.Timestamp, pd.Timestamp]]]]:
    """Apply synthetic gaps to target and/or neighbor.

    Parameters
    ----------
    target, neighbor : time-aligned inputs
    spec : GapSpec

    Returns
    -------
    (target_gapped, neighbor_gapped, gap_windows)
        - `gap_windows` provides timestamp ranges actually masked under keys
          'target' and possibly 'neighbor'.
    """
    idx = target.index
    n = len(idx)
    gaps_t = _choose_gap_windows(n, spec)

    def _mask_series(s: pd.Series, windows: List[Tuple[int, int]]):
        m = s.copy()
        for (i0, i1) in windows:
            m.iloc[i0:i1] = np.nan
        return m

    # Decide neighbor gaps
    if spec.strategy == GapStrategy.TARGET_ONLY:
        gaps_n: List[Tuple[int, int]] = []
    elif spec.strategy == GapStrategy.BOTH:
        gaps_n = gaps_t
    else:  # STAGGERED
        # Draw new windows for neighbor
        gaps_n = _choose_gap_windows(n, spec)

    t_gap = _mask_series(target, gaps_t)
    if isinstance(neighbor, pd.DataFrame):
        n_gap = neighbor.copy()
        for col in n_gap.columns:
            n_gap[col] = _mask_series(n_gap[col], gaps_n)
    else:
        n_gap = _mask_series(neighbor, gaps_n)

    # Build human-readable windows with timestamps
    to_ts = lambda w: (idx[w[0]], idx[min(len(idx)-1, w[1]-1)])
    gap_dict = {
        "target": [to_ts(w) for w in gaps_t],
        "neighbor": [to_ts(w) for w in gaps_n],
    }
    return t_gap, n_gap, gap_dict

def run_gapfill_experiment(
    data_provider: Callable[[], Tuple[pd.Series, pd.Series | pd.DataFrame, Optional[pd.Series]]],
    gap_spec: GapSpec,
    methods: Iterable[str] = ("ols", "lagged_reg", "state_space"),
    lags: Optional[Iterable[int] | int] = None,
    resample: Optional[str] = None,
    bounds: Tuple[Optional[float], Optional[float]] = (None, None),
    arimax_order: Tuple[int, int, int] = (1, 0, 0),
) -> Dict[str, object]:
    """Create synthetic gaps, run selected methods, and collect results.

    Returns a dict suitable for `plot_imputations`.
    """
    target, neighbor, regime = data_provider()
    # Optionally regularize frequency for both
    if resample is not None:
        target = target.resample(resample).mean()
        if isinstance(neighbor, pd.DataFrame):
            neighbor = neighbor.resample(resample).mean()
        else:
            neighbor = neighbor.resample(resample).mean()
        if regime is not None:
            regime = regime.resample(resample).ffill()

    # Basic alignment (outer to preserve all timestamps); we'll rely on the filler for final alignment
    target, neighbor = target.align(neighbor, join="outer")

    # Apply gaps
    target_g, neighbor_g, gap_windows = apply_gaps(target, neighbor, gap_spec)

    # Run methods
    results: Dict[str, Dict] = {}
    for m in methods:
        out = fill_from_neighbor(
            target=target_g,
            neighbor=neighbor_g,
            method=m,
            regime=regime,
            lags=lags,
            resample=None,
            bounds=bounds,
            arimax_order=arimax_order,
        )
        results[m] = out

    return {
        "target": target,
        "neighbor": neighbor,
        "target_gapped": target_g,
        "neighbor_gapped": neighbor_g,
        "gap_windows": gap_windows,
        "results": results,
    }

def plot_imputations(
    target: pd.Series,
    target_gapped: pd.Series,
    results: Dict[str, Dict],
    neighbor: pd.Series | pd.DataFrame | None = None,
    show_pi: bool = True,
    tlim: Optional[Tuple[pd.Timestamp, pd.Timestamp]] = None,
    title: Optional[str] = None,
):
    """Plot original vs gapped target and method imputations.

    Parameters
    ----------
    target : original complete (or less-gappy) series
    target_gapped : with synthetic gaps
    results : mapping method -> dict returned by `fill_from_neighbor`
    neighbor : optional neighbor series to plot lightly for context
    show_pi : draw prediction intervals when available
    tlim : optional (t0, t1) to focus x-limits
    title : overall title
    """
    fig, ax = plt.subplots(figsize=(12, 5))

    # Plot original and gapped
    target.plot(ax=ax, linewidth=1.5, label="target (true)")
    target_gapped.plot(ax=ax, linewidth=1.0, linestyle=":", label="target (with gaps)")

    # Each method overlay
    for m, out in results.items():
        yhat = pd.Series(out["yhat"])  # ensure Series
        yhat.name = f"{m} yhat"
        ax.plot(yhat.index, yhat.values, linewidth=1.4, label=f"{m} (imputed)")
        if show_pi and out.get("pi_lower") is not None:
            pil = pd.Series(out["pi_lower"])  # type: ignore
            piu = pd.Series(out["pi_upper"])  # type: ignore
            ax.fill_between(pil.index, pil.values, piu.values, alpha=0.12)

    if neighbor is not None:
        # Plot neighbor faintly on secondary axis for reference
        ax2 = ax.twinx()
        if isinstance(neighbor, pd.DataFrame):
            neighbor.sum(axis=1).plot(ax=ax2, alpha=0.3, linewidth=0.8, color="gray", label="neighbor (sum)")
        else:
            neighbor.plot(ax=ax2, alpha=0.3, linewidth=0.8, color="gray", label="neighbor")
        ax2.grid(False)
        ax2.set_ylabel("neighbor (scaled)")

    ax.set_xlabel("time")
    ax.set_ylabel("value")
    if title:
        ax.set_title(title)
    ax.legend(loc="upper left", ncol=2)
    if tlim is not None:
        ax.set_xlim(tlim)
    ax.grid(True, alpha=0.3)
    fig.tight_layout()
    return fig

def _synthetic_provider(n=1000, freq="H", lag=3, noise=0.1):
    idx = pd.date_range("2020-01-01", periods=n, freq=freq)
    t = np.arange(n)
    # Base signal with slow seasonal-ish component
    base = np.sin(2 * np.pi * t / 240.0) + 0.2 * np.sin(2 * np.pi * t / 24.0)
    x = pd.Series(base + 0.05 * np.random.randn(n), index=idx, name="neighbor")
    y = pd.Series(0.5 + 1.2 * x.shift(lag).bfill() + noise * np.random.randn(n), index=idx, name="target")

    return y, x, None

def _maybe_run_demo():
    try:
        # Try user's provider
        _ = provide_example_data
        # Accessing the function succeeds; attempt call
        y, x, reg = provide_example_data()
    except Exception:
        # Fallback to synthetic
        y, x, reg = _synthetic_provider()

    gap_spec = GapSpec(n_gaps=6, min_len=12, max_len=60, seed=7, strategy=GapStrategy.TARGET_ONLY)
    out = run_gapfill_experiment(
        data_provider=lambda: (y, x, reg),
        gap_spec=gap_spec,
        methods=["ols", "lagged_reg", "state_space"],
        lags=range(0, 4),
        resample=None,
    )
    fig = plot_imputations(
        target=out["target"],
        target_gapped=out["target_gapped"],
        results=out["results"],
        neighbor=out["neighbor"],
        show_pi=True,
        title="Neighbor fill: synthetic demo (falls back if no provider)",
    )
    plt.show()




if __name__ == "__main__":
    _maybe_run_demo()
